@ApiOperation("根据id查询章节")
@GetMapping("get/{id}")
public R getById(
    @ApiParam(value="章节id", required = true)
    @PathVariable String id){
    Chapter chapter = chapterService.getById(id);
    if (chapter != null) {
        return R.ok().data("item", chapter);
    } else {
        return R.error().message("数据不存在");
    }
}
